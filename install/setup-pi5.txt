    # Try to find an existing wayland socket in /run/user/$UID/
    if ls /run/user/$UID/wayland-* 1> /dev/null 2>&1; then
        export WAYLAND_DISPLAY=$(ls /run/user/$UID/wayland-* | head -n 1 | xargs basename)
    fi
fi
