// rotate screen
mkdir -p ~/.config/autostart
vi ~/.config/autostart/screen-rotation.desktop
[Desktop Entry]
Type=Application
Name=Screen Rotation
Exec=sh -c "sleep 3 && /usr/bin/wlr-randr --output DSI-1 --transform 270"
Terminal=false
X-GNOME-Autostart-enabled=true

// wf-panel
[panel]
# Set this to your second screen name (e.g., HDMI-A-1)
output=HDMI-A-1
position=top
icon_size=32
window-list_max_width=200
monitor=HDMI-A-1
